20 double GeoMetricMean(ActField const &act_field, int **sigma, PixelPos pos)
21 {
22 // NO BOUNDS CHECK for sigma
23 int mainspin = sigma[pos.x][pos.y];
24 double output = 1.0;
25 int count = 0;
26 for (int i = -1; i <= 1; i++)
27 for (int j = -1; j <= 1; j++)
28 {
29 PixelPos neighbour_pos(i, j);
30 neighbour_pos += pos;
31 if (sigma[neighbour_pos.x][neighbour_pos.y] == mainspin)
32 {
33 double value = act_field.Value(neighbour_pos);
34 if (value <= 0.0)
35 {
36 return 0.0;
37 }
38 count++;
39 output *= value;
40 }
41 }
42 if (count == 0.0)
43 return 0.0;
44 return std::pow(output, 1.0 / count);
45 }
46}

91double ACT::DeltaH(ActField const &act_field, int **sigma, PixelPos from,
92 PixelPos to, double const lambda_act, double const max_Act)
93{
94 double GM_source = GeoMetricMean(act_field, sigma, from);
95
96 // If medium has positive act, something went wrong somewhere. It is catched here.
97 if (sigma[from.x][from.y] == 0 && GM_source > 0)
98 throw std::runtime_error("from medium has positive act!!");
99
100 double GM_target = GeoMetricMean(act_field, sigma, to);
101 if (sigma[to.x][to.y] == 0 && GM_target > 0)
102 throw std::runtime_error("to medium has positive act!!");
103
104 return (lambda_act / max_Act) * (GM_source - GM_target);
105}
